The destructure and extract and assign value to 3 objects according to below prisma models 
```
model Staff {
  id            Int    @id @default(autoincrement())
  StaffName     String
  AgentName     String
  StaffCategory String
  Department    String
  PostUnit      String
  ManagerName   String
  ManagerTitle  String
  ManagerEmail  String}


model StaffContract {
  id                Int      @id @default(autoincrement())
  ContractStartDate DateTime
  ContractEndDate   DateTime
  AnnualLeave       Int
  IsActive Boolean @default(false) // New field to indicate the active contract
}

model User {
  id            String      @id @default(cuid())
  username      String?     @unique
  name          String? <--use staffname
  email         String?     @unique
 
 
  password      String
}
```
